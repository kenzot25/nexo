/nexo                                             # full pipeline on current directory → Obsidian vault
/nexo <path>                                      # full pipeline on specific path
/nexo <path> --mode deep                          # thorough extraction, richer INFERRED edges
/nexo <path> --update                             # incremental - re-extract only new/changed files
/nexo <path> --directed                            # build directed graph (preserves edge direction: source→target)
/nexo <path> --cluster-only                       # rerun clustering on existing graph
/nexo <path> --no-viz                             # skip visualization, just report + JSON
/nexo <path> --html                               # (HTML is generated by default - this flag is a no-op)
/nexo <path> --obsidian --obsidian-dir ~/vaults/my-project  # write vault to custom path (e.g. existing vault)
/nexo <path> --watch                              # watch folder, auto-rebuild on code changes (no LLM needed)
/nexo <path> --watch --obsidian-sync              # also refresh Obsidian vault after code rebuild
/nexo <path> --watch --obsidian-sync --obsidian-dir ~/vaults/my-project  # custom Obsidian vault path for watch sync
/nexo add <url>                                   # fetch URL, save to ./raw, update graph
/nexo add <url> --author "Name"                   # tag who wrote it
/nexo add <url> --contributor "Name"              # tag who added it to the corpus
/nexo query "<question>"                          # BFS traversal - broad context
/nexo query "<question>" --dfs                    # DFS - trace a specific path
/nexo query "<question>" --budget 1500            # cap answer at N tokens
/nexo path "AuthModule" "Database"                # shortest path between two concepts
/nexo explain "SwinTransformer"                   # plain-language explanation of a node
